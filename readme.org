* aria2c config

1. brew install aria2c && download Aria2GUI

2. config file
	~/aria2c.config

3. 配置
   #+BEGIN_SRC python
   aria () {
	    local option=$1
	    case $option in
		-s) aria2c --conf-path="/Users/yingdai/aria2.conf" -D
		    open -a Aria2GUI
		    ;;
		-k) killall -2 aria2c
		    killall Aria2GUI
		    ;;
		*)  echo "aria -s : start service"
		    echo "aria -k : kill service"
	    esac
   }
   #+END_SRC
